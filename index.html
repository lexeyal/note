связь медленная переделал скрипт архивирования

#!/bin/sh
# v1.0
datestamp=`date +'%Y-%m-%d'`
whatp='/home/********/'
whatf='*******'
log="./log.txt"

##
echo `date` >> $log
echo "start sync" >> $log
rsync --progress --delete -aogvze "ssh -p ****" **********@*********:$whatp$whatf ./
$whatf ./
echo `date` >> $log
echo "start tgz" >> $log
tar -czf ./$datestamp-$whatf.tgz ./$whatf
echo `date` >> $log
echo "end" >> $log
echo "===================================" >> $log

exit 0
